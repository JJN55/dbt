from pyspark.sql import SparkSession
import re

def convert_to_underscore(text):
    """
    Converts whitespace and special characters in a string to underscores.
    Args:
      text: The input string.
    Returns:
      The modified string with whitespace and special characters replaced by underscores.
    """
    # Replace all non-alphanumeric characters (including whitespace and special characters)
    # with a single underscore.
    # The '+' ensures multiple consecutive non-alphanumeric characters are replaced by one underscore.
    cleaned_text = re.sub(r'[^a-zA-Z0-9_]+', '_', text)
    return cleaned_text

def source_select(text):
    """
    Replaces any character that is NOT:
    letters (a-z, A-Z)
    numbers (0-9)
    underscore (_)
    With an underscore _.
    """
    cleaned_text = re.sub(r'[^a-zA-Z0-9_]+', '_', text)
    if  (cleaned_text == text):
        return cleaned_text
    else:
        return f"{{{{adapter.quote(\"{text}\")}}}} as {cleaned_text}"
